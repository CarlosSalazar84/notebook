
\paragraph{Combinaci\'on (Coeficiente Binomial): N\'umero de subconjuntos de k elementos escogidos de un conjunto con n elementos\\ \\}
{\LARGE $ \binom{n}{k} = \binom{n}{n-k} = \frac{n!}{k!(n-k)!} $}

\paragraph{Combinaci\'on con repetici\'on: N\'umero de grupos formados por n elementos, partiendo de m tipos de elementos.\\ \\}


{\LARGE $ CR_{m}^{n} = \binom{m+n-1}{n} = \frac{(m + n - 1)!}{n!(m-1)!} $}

\paragraph{Permutaci\'on: N\'umero de formas de agrupar n elementos, donde importa el orden y sin repetir elementos\\ \\}

{\LARGE $ P_{n} = n! $}

\paragraph{Elegir r elementos de n posibles con repetici\'on\\ \\}

{\LARGE $ n^{r} $}

\paragraph{Permutaciones con repetici\'on: Se tienen n elementos donde el primer elemento se repite a veces , el segundo b veces , el tercero c veces, ...\\ \\}

{\LARGE $ PR_{n}^{a,b,c...} = \frac{P_{n}}{a!b!c!...}$}

\paragraph{Permutaciones sin repetici\'on: N\'umero de formas de agrupar r elementos de n disponibles, sin repetir elementos \\ \\}

{\LARGE $ \frac{n!}{(n-r)!}$}
